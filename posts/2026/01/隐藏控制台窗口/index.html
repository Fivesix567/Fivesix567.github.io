    #include <windows.h>

int main() {
    // 获取控制台窗口句柄
    HWND hWnd = GetConsoleWindow();
    // 隐藏
    ShowWindow(hWnd, SW_HIDE);
    
    MessageBoxA(0, "弹窗！", "Test", 0);
    return 0;
}
